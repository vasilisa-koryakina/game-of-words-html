<!--работаем с PyScript — фреймворком для фронтенда (того кода, который обрабатывает браузер на стороне клиента,
то есть на пользовательском устройстве) на Python-->
<HTML> <!--запускаем html код - тот, который может расшифровать программа "браузер"*/
    <head><!--в head описываем настройки для документа-->
     <link rel="stylesheet" href="https://pyscript.net/alpha/pyscript.css" /> <!--нужно указать в нём ссылку на
     PyScript, чтобы получить доступ к интерфейсу PyScript-->
     <script defer src="https://pyscript.net/alpha/pyscript.js"></script> <!--тоже ссылка на pyscript-->
     
    </head>
    <body> <!--в body описываем то, что будет отображено на странице-->
        <div>Введите слово</div><!--просто отображаем текст-->
        <button id="submit-button" type="submit" pys-onClick="my_function">Нажми</button><!--создаём кнопку, привязываем
        к pys-onClick функцию, которую написали на python -->
        <input type="text" id="test-input" style="background-color:red; width: 200px" placeholder="место для первого слова" >
        <input type="text" id="test-input2" style="background-color: blue; width: 200px" placeholder="место для остальных слов">
        <!--выше два поля ввода для получения информации от пользователя. с помощью id задаём им имена, чтобы потом на
        них ссылаться и получать информацию о них -->
        <div id="test-output"></div><!--блок для вывода информации, ответа от программы, после нажатия кнопки и обработки
        с помощью питона, введённой пользователем информации-->
        <!--здесь, внутри открывающего и закрывающего тегов <py-script></py-script> код питона, адаптированный
      для браузера -->
      
      <py-script> 
        from js import console #<!--импортируем библиотеку и необходимую константу для создания окна и вывода результата
            #на экран, как на консоль-->
  
        def my_function(*args, **kwargs):#<!--создаём функцию, которая будет принимать любые аргументы-->
                        
            console.log(f'args: {args}')#<!--выводит текст в консоль в виде сообщения журнала,
            #без этого не получится вывести всё на экран, так надо обработать, принимаемые функцией аргументы-->
            console.log(f'kwargs: {kwargs}')
            text = Element('test-input').element.value #<!--присваиваем переменной то, что берём из поля ввода,
            #используем id (уникальный идентификатор) поля ввода, который присволи ему сами -->
            text2 = Element('test-input2').element.value
            text3 = "Верно" 
            text4 = "Неверно"
            console.log(f'text: {text}') #<!--все переменные надо обработать таким образом для вывода -->
            console.log(f'text2: {text2}')
            console.log(f'text3: {text3}')
            console.log(f'text4: {text4}')
            if text[-1] == text2[0]: #<!--сравниваем, последний символ первого слова и первый символ второго слова-->
                Element('test-output').element.innerText = text3 #<!--в случае верности сравнения, выводим "верно"-->
            else: #<!-- в ином случае выводим "неверно" -->
                Element('test-output').element.innerText = text4

     </py-script> </body> <!-- закрываем теги html, многие из них требуют закрывающий тег, который такой же, но со 
     слешем (/)-->
     </html>
